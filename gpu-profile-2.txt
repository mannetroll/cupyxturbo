|   487 |        |       |       |def dns_pao_host_init(S: DnsState):                                                                        |
|   488 |        |       |       |    xp = S.xp                                                                                              |
|   489 |        |       |       |    N = S.NX                                                                                               |
|   490 |        |       |       |    NE = S.NZ                                                                                              |
|   491 |        |       |       |    ND2 = N // 2                                                                                           |
|   492 |        |       |       |    NED2 = NE // 2                                                                                         |
|   493 |        |       |       |    PI = np.float32(3.14159265358979)                                                                      |
|   494 |        |       |       |                                                                                                           |
|   495 |        |       |       |    DXZ = np.float32(2.0) * PI / np.float32(N)                                                             |
|   496 |        |       |       |    K0 = np.float32(S.K0)                                                                                  |
|   497 |        |       |       |    NORM = PI * K0 * K0                                                                                    |
|   498 |        |       |       |                                                                                                           |
|   499 |        |       |       |    print("--- INITIALIZING SciPy/CuPy ---", _dt.datetime.now().strftime("%Y-%m-%d %H:%M"))                |
|   500 |        |       |       |    print(f" N={N}, K0={int(K0)}, Re={S.Re}")                                                              |
|   501 |        |       |       |                                                                                                           |
|   502 |        |       |       |    # ------------------------------------------------------------------                                   |
|   503 |        |       |       |    # Build ALFA(N/2) and GAMMA(N)  (Fortran DALFA, DGAMMA, E1, E3)                                        |
|   504 |        |       |       |    # ------------------------------------------------------------------                                   |
|   505 |        |       |       |    alfa = np.zeros(ND2, dtype=np.float32)                                                                 |
|   506 |        |       |       |    gamma = np.zeros(NE, dtype=np.float32)                                                                 |
|   507 |        |       |       |                                                                                                           |
|   508 |        |       |       |    E1 = np.float32(1.0)                                                                                   |
|   509 |        |       |       |    E3 = np.float32(1.0) / E1                                                                              |
|   510 |        |       |       |                                                                                                           |
|   511 |        |       |       |    DALFA = np.float32(1.0) / E1                                                                           |
|   512 |        |       |       |    DGAMMA = np.float32(1.0) / E3                                                                          |
|   513 |        |       |       |                                                                                                           |
|   514 |        |       |       |    for x in range(NED2):                                                                                  |
|   515 |        |       |       |        alfa[x] = np.float32(x) * DALFA                                                                    |
|   516 |        |       |       |                                                                                                           |
|   517 |        |       |       |    gamma[0] = np.float32(0.0)                                                                             |
|   518 |        |       |       |    for z in range(1, NED2 + 1):                                                                           |
|   519 |        |       |       |        gamma[z] = np.float32(z) * DGAMMA                                                                  |
|   520 |        |       |       |        gamma[NE - z] = -gamma[z]                                                                          |
|   521 |        |       |       |                                                                                                           |
|   522 |        |       |       |    # ------------------------------------------------------------------                                   |
|   523 |        |       |       |    # Host spectral UR: complex field UR(kx,z,comp)                                                        |
|   524 |        |       |       |    # comp=0 â†’ u1, comp=1 â†’ u3 (Fortran components 1 and 2)                                            |
|   525 |        |       |       |    #                                                                                                      |
|   526 |        |       |       |    #   UR[x,z,c]  where  x âˆˆ [0..ND2-1], z âˆˆ [0..NE-1], c âˆˆ {0,1}                                   |
|   527 |        |       |       |    # ------------------------------------------------------------------                                   |
|   528 |        |       |       |    UR = np.zeros((ND2, NE, 2), dtype=np.complex64)                                                        |
|   529 |        |       |       |                                                                                                           |
|   530 |        |       |       |    # ------------------------------------------------------------------                                   |
|   531 |        |       |       |    # Fortran random vector RANVEC(97)                                                                     |
|   532 |        |       |       |    # ------------------------------------------------------------------                                   |
|   533 |        |       |       |    seed = [int(S.seed_init)]  # mimics ISEED SAVE                                                         |
|   534 |        |       |       |    RANVEC = np.zeros(97, dtype=np.float32)                                                                |
|   535 |        |       |       |                                                                                                           |
|   536 |        |       |       |    # "warm-up" 97 calls                                                                                   |
|   537 |        |       |       |    for _ in range(97):                                                                                    |
|   538 |        |       |       |        frand(seed)                                                                                        |
|   539 |        |       |       |                                                                                                           |
|   540 |        |       |       |    # fill RANVEC                                                                                          |
|   541 |        |       |       |    for i in range(97):                                                                                    |
|   542 |        |       |       |        RANVEC[i] = frand(seed)                                                                            |
|   543 |        |       |       |                                                                                                           |
|   544 |        |       |       |    def random_from_vec(r: np.float32) -> np.float32:                                                      |
|   545 |        |       |       |        idx = int(float(r) * 97.0)                                                                         |
|   546 |        |       |       |        if idx < 0:                                                                                        |
|   547 |        |       |       |            idx = 0                                                                                        |
|   548 |        |       |       |        if idx > 96:                                                                                       |
|   549 |        |       |       |            idx = 96                                                                                       |
|   550 |        |       |       |        v = RANVEC[idx]                                                                                    |
|   551 |        |       |       |        RANVEC[idx] = r                                                                                    |
|   552 |        |       |       |        return v                                                                                           |
|   553 |        |       |       |                                                                                                           |
|   554 |        |       |       |    # ------------------------------------------------------------------                                   |
|   555 |        |       |       |    # Generate isotropic random spectrum (Fortran DO 500/510 loops)                                        |
|   556 |        |       |       |    # ------------------------------------------------------------------                                   |
|   557 |        |       |       |    print("Generate isotropic random spectrum...")                                                         |
|   558 |        |       |       |    for z in range(NE):                                                                                    |
|   559 |        |       |       |        if z % 1000 == 0:                                                                                  |
|   560 |        |       |       |            print(f"z={z}/{NE}")                                                                           |
|   561 |        |       |       |                                                                                                           |
|   562 |        |       |       |        gz = gamma[z]                                                                                      |
|   563 |        |       |       |        for x in range(NED2):                                                                              |
|   564 |        |       |       |            r = frand(seed)                                                                                |
|   565 |        |       |       |            th = np.float32(2.0) * PI * random_from_vec(r)                                                 |
|   566 |        |       |       |                                                                                                           |
|   567 |        |       |       |            ARG = np.complex64(np.cos(th) + 1j * np.sin(th))                                               |
|   568 |        |       |       |                                                                                                           |
|   569 |        |       |       |            ax = alfa[x]                                                                                   |
|   570 |        |    2% |       |            K2 = np.float32(ax * ax + gz * gz)                                                             |
|   571 |        |       |       |            K = np.float32(np.sqrt(K2)) if K2 > 0.0 else np.float32(0.0)                                   |
|   572 |        |       |       |                                                                                                           |
|   573 |        |       |       |            if ax == 0.0:                                                                                  |
|   574 |        |       |       |                # ALFA(X) == 0: purely u1 mode                                                             |
|   575 |        |       |       |                UR[x, z, 1] = np.complex64(0.0 + 0.0j)                                                     |
|   576 |        |       |       |                                                                                                           |

|   578 |        |       |       |                amp = np.float32(np.sqrt(ABSU2))                                                           |
|   579 |        |       |       |                UR[x, z, 0] = np.complex64(amp) * ARG                                                      |
|   580 |        |       |       |            else:                                                                                          |
|   581 |        |       |       |                denom = np.float32(1.0) + (gz * gz) / (ax * ax)                                            |
|   582 |        |       |       |                ABSW2 = np.float32(np.exp(- (K / K0) * (K / K0)) / (denom * NORM))                         |
|   583 |        |       |       |                ampw = np.float32(np.sqrt(ABSW2))                                                          |
|   584 |        |       |       |                                                                                                           |
|   585 |        |       |       |                w = np.complex64(ampw) * ARG                                                               |
|   586 |        |       |       |                u = np.complex64(- (gz / ax)) * w  # -GAMMA/ALFA * UR(.,.,2)                               |
|   587 |        |       |       |                                                                                                           |
|   588 |        |       |       |                UR[x, z, 1] = w                                                                            |
|   589 |        |       |       |                UR[x, z, 0] = u                                                                            |
|   590 |        |       |       |                                                                                                           |

|   778 |        |       |       |                                                                                                           |
|   779 |        |       |       |                                                                                                           |
|   780 |        |       |       |def vfft_full_forward_ur_full_to_uc_full(S: DnsState) -> None:                                             |
|   781 |        |       |       |    """                                                                                                    |
|   782 |        |       |       |    UR_full (3, NZ_full, NX_full) â†’ UC_full (3, NZ_full, NK_full)                                        |

|   787 |        |       |       |                                                                                                           |

|   791 |        |       |       |    UR = S.ur_full                                                                                         |
|   792 |        |       |       |    fft = S.fft                                                                                            |
|   793 |        |       |       |    if fft is None:                                                                                        |
|   794 |        |       |       |        raise RuntimeError("scipy.fft is not available on CPU (import scipy.fft failed).")                 |

|   799 |        |       |       |    else:                                                                                                  |
|   800 |        |       |       |        plan = S.fft_plan_rfft2_ur_full                                                                    |
|   801 |        |       |       |        if plan is not None:                                                                               |
|   802 |        |       |       |            with plan:                                                                                     |
|   803 |        |       |   2%  |                UC = fft.rfft2(UR, s=(S.NZ_full, S.NX_full), axes=(1, 2))                                  |
|   804 |        |       |       |        else:                                                                                              |
|   805 |        |       |       |            UC = fft.rfft2(UR, s=(S.NZ_full, S.NX_full), axes=(1, 2))                                      |
|   806 |        |       |       |                                                                                                           |
|   807 |        |       |       |    # Assign back; uc_full is complex64, assignment will down-cast if needed                               |
|   808 |        |       |       |    S.uc_full[...] = UC                                                                                    |
|   809 |        |       |       |                                                                                                           |
|   810 |        |       |       |                                                                                                           |
|   811 |        |       |       |# ---------------------------------------------------------------------------                              |
|   812 |        |       |       |# CALCOM â€” spectral vorticity from UC_full (dnsCudaCalcom)                                               |
|   813 |        |       |       |# ---------------------------------------------------------------------------                              |
|   814 |        |       |       |                                                                                                           |
|   815 |        |       |       |def dns_calcom_from_uc_full(S: DnsState) -> None:                                                          |
|   816 |        |       |       |    """                                                                                                    |
|   817 |        |       |       |    Python/xp port of dnsCudaCalcom:                                                                       |
|   818 |        |       |       |                                                                                                           |
|   819 |        |       |       |      OM2(ix,iz) = i * [ GAMMA(iz)*UC1(ix,iz) - ALFA(ix)*UC2(ix,iz) ]                                      |
|   820 |        |       |       |                                                                                                           |
|   821 |        |       |       |    Uses:                                                                                                  |
|   822 |        |       |       |      S.uc_full : (3, NZ_full, NK_full)  [comp,z,kx]                                                       |
|   823 |        |       |       |      S.alfa    : (NX_half,)                                                                               |
|   824 |        |       |       |      S.gamma   : (NZ,)                                                                                    |
|   825 |        |       |       |    Writes:                                                                                                |
|   826 |        |       |       |      S.om2     : (NZ, NX_half)                                                                            |
|   827 |        |       |       |    """                                                                                                    |
|   828 |        |       |       |    xp = S.xp                                                                                              |
|   829 |        |       |       |                                                                                                           |
|   830 |        |       |       |    Nbase = int(S.Nbase)                                                                                   |
|   831 |        |       |       |    NX_full = int(S.NX_full)                                                                               |
|   832 |        |       |       |    NZ_full = int(S.NZ_full)                                                                               |
|   833 |        |       |       |    NK_full = int(S.NK_full)                                                                               |
|   834 |        |       |       |                                                                                                           |
|   835 |        |       |       |    NX_half = Nbase // 2                                                                                   |
|   836 |        |       |       |    NZ = Nbase                                                                                             |
|   837 |        |       |       |                                                                                                           |
|   838 |        |       |       |    alfa_1d = S.alfa.astype(xp.float32)      # (NX_half,)                                                  |
|   839 |        |       |       |    gamma_1d = S.gamma.astype(xp.float32)     # (NZ,)                                                      |
|   840 |        |       |       |                                                                                                           |
|   841 |        |       |       |    # UC_full layout: [comp, z, kx]                                                                        |
|   842 |        |       |       |    uc1_full = S.uc_full[0]                   # (NZ_full, NK_full)                                         |
|   843 |        |       |       |    uc2_full = S.uc_full[1]                   # (NZ_full, NK_full)                                         |
|   844 |        |       |       |                                                                                                           |
|   845 |        |       |       |    # We only use the first NZ rows and NX_half kx-modes                                                   |
|   846 |        |       |       |    uc1 = uc1_full[:NZ, :NX_half]             # (NZ, NX_half)                                              |
|   847 |        |       |       |    uc2 = uc2_full[:NZ, :NX_half]             # (NZ, NX_half)                                              |
|   848 |        |       |       |                                                                                                           |
|   849 |        |       |       |    ax = alfa_1d[None, :]                     # (1, NX_half)                                               |
|   850 |        |       |       |    gz = gamma_1d[:, None]                    # (NZ, 1)                                                    |
|   851 |        |       |       |                                                                                                           |
|   852 |        |       |       |    # diff = GAMMA*UC1 - ALFA*UC2                                                                          |
|   853 |        |       |       |    diff = gz * uc1 - ax * uc2                # (NZ, NX_half), complex                                     |
|   854 |        |       |       |                                                                                                           |
|   855 |        |       |       |    # om = i * diff = (-Im(diff), Re(diff))                                                                |
|   856 |        |       |       |    diff_r = diff.real                                                                                     |
|   857 |        |       |       |    diff_i = diff.imag                                                                                     |
|   858 |        |       |       |                                                                                                           |
|   859 |        |       |       |    om_r = -diff_i                                                                                         |
|   860 |        |       |       |    om_i = diff_r                                                                                          |
|   861 |        |       |       |                                                                                                           |
|   862 |        |       |       |    S.om2[...] = xp.asarray(om_r + 1j * om_i, dtype=xp.complex64)                                          |
|   863 |        |       |       |                                                                                                           |
|   864 |        |       |       |                                                                                                           |
|   865 |        |       |       |# ---------------------------------------------------------------------------                              |
|   866 |        |       |       |# STEP2B â€” build uiuj and forward FFT (dnsCudaStep2B)                                                    |
|   867 |        |       |       |# ---------------------------------------------------------------------------                              |
|   868 |        |       |       |def dns_step2b(S: DnsState) -> None:                                                                       |
|   869 |        |       |       |    """                                                                                                    |
|   870 |        |       |       |    Python/CuPy port of dnsCudaStep2B(DnsDeviceState *S).                                                  |
|   871 |        |       |       |                                                                                                           |
|   872 |        |       |       |    Mirrors Fortran STEP2B:                                                                                |
|   873 |        |       |       |                                                                                                           |
|   874 |        |       |       |      1) Build uiuj in UR(x,z,1..3) on the full 3/2 grid                                                   |
|   875 |        |       |       |      2) Full-grid forward FFT: UR_full â†’ UC_full (3 components)                                         |
|   876 |        |       |       |         (VRFFTF + VCFFTF in Fortran)                                                                      |
|   877 |        |       |       |      3) Zero UC(X,NZ+1,I) for X<=NX/2, I=1..3                                                             |
|   878 |        |       |       |    """                                                                                                    |
|   879 |        |       |       |    xp = S.xp                                                                                              |
|   880 |        |       |       |                                                                                                           |
|   881 |        |       |       |    # Geometry on the full 3/2 grid                                                                        |
|   882 |        |       |       |    N = S.Nbase          # NX = NZ = Nbase (Fortran NX,NZ)                                                 |
|   883 |        |       |       |    NX_full = S.NX_full        # 3*N/2                                                                     |
|   884 |        |       |       |    NZ_full = S.NZ_full        # 3*N/2                                                                     |
|   885 |        |       |       |    NK_full = S.NK_full        # 3*N/4+1                                                                   |
|   886 |        |       |       |                                                                                                           |
|   887 |        |       |       |    UR = S.ur_full                                                                                         |
|   888 |        |       |       |    UC = S.uc_full                                                                                         |
|   889 |        |       |       |                                                                                                           |
|   890 |        |       |       |    u = UR[0]   # (NZ_full, NX_full)                                                                       |
|   891 |        |       |       |    w = UR[1]   # (NZ_full, NX_full)                                                                       |
|   892 |        |       |       |                                                                                                           |
|   893 |        |       |       |    # Use in-place multiplies to avoid temporaries                                                         |
|   894 |        |       |       |    xp.multiply(u, w, out=UR[2])  # u * w                                                                  |
|   895 |        |       |   2%  |    xp.multiply(u, u, out=UR[0])  # u^2                                                                    |
|   896 |        |       |   2%  |    xp.multiply(w, w, out=UR[1])  # w^2                                                                    |
|   897 |        |       |       |                                                                                                           |
|   898 |        |       |       |    vfft_full_forward_ur_full_to_uc_full(S)                                                                |
|   899 |        |       |       |                                                                                                           |
|   900 |        |       |       |    NX_half = N // 2                                                                                       |
|   901 |        |       |       |    NZ = N                                                                                                 |
|   902 |        |       |       |    z_mid = NZ                                                                                             |
|   903 |        |       |       |                                                                                                           |
|   904 |        |       |       |    kx_max = min(NX_half, NK_full)                                                                         |
|   905 |        |       |       |                                                                                                           |
|   906 |        |       |       |    if z_mid < NZ_full and kx_max > 0:                                                                     |
|   907 |        |       |       |        UC[0:3, z_mid, 0:kx_max] = xp.complex64(0.0 + 0.0j)                                                |
|   908 |        |       |       |                                                                                                           |
|   909 |        |       |       |                                                                                                           |
|   910 |        |       |       |# ---------------------------------------------------------------------------                              |
|   911 |        |       |       |# STEP3 â€” vorticity update using om2 & fnm1                                                              |
|   912 |        |       |       |# ---------------------------------------------------------------------------                              |
|   913 |        |       |       |def dns_step3(S: DnsState) -> None:                                                                        |
|   914 |        |       |       |    xp = S.xp                                                                                              |
|   915 |        |       |       |                                                                                                           |
|   916 |        |       |       |    om2 = S.om2                                                                                            |
|   917 |        |       |       |    fnm1 = S.fnm1                                                                                          |
|   918 |        |       |       |    alfa = S.alfa                                                                                          |

|   920 |        |       |       |    uc_full = S.uc_full                                                                                    |
|   921 |        |       |       |                                                                                                           |
|   922 |        |       |       |    Nbase = int(S.Nbase)                                                                                   |
|   923 |        |       |       |    NX_half = Nbase // 2                                                                                   |

|   929 |        |       |       |    cnm1 = xp.float32(S.cnm1)                                                                              |
|   930 |        |       |       |                                                                                                           |
|   931 |        |       |       |    z_spec = S.step3_z_spec                                                                                |
|   932 |        |       |       |    divxz = S.step3_divxz                                                                                  |
|   933 |        |       |       |    GA = S.step3_GA                                                                                        |
|   934 |        |       |       |    G2mA2 = S.step3_G2mA2                                                                                  |
|   935 |        |       |       |    K2 = S.step3_K2                                                                                        |
|   936 |        |       |       |                                                                                                           |
|   937 |        |       |       |    uc0_low = uc_full[0, :, :NX_half]                                                                      |

|   946 |        |       |       |    xp.take(uc2_low, z_spec, axis=0, out=uc3_th)                                                           |
|   947 |        |       |       |                                                                                                           |
|   948 |        |       |       |    tmp_FN = S.scratch1                                                                                    |
|   949 |        |       |       |    tmp_c = S.scratch2                                                                                     |
|   950 |        |       |       |    xp.subtract(uc1_th, uc2_th, out=tmp_FN)                                                                |

|   953 |        |       |       |    xp.add(tmp_FN, tmp_c, out=tmp_FN)                                                                      |
|   954 |        |       |       |    tmp_FN *= divxz                                                                                        |
|   955 |        |       |       |                                                                                                           |
|   956 |        |       |       |    VT = xp.float32(0.5) * visc * dt                                                                       |
|   957 |        |       |       |    ARG = S.step3_ARG                                                                                      |
|   958 |        |       |       |    DEN = S.step3_DEN                                                                                      |
|   959 |        |       |       |    xp.multiply(K2, VT, out=ARG)                                                                           |
|   960 |        |       |       |    xp.add(ARG, xp.float32(1.0), out=DEN)                                                                  |
|   961 |        |       |       |                                                                                                           |
|   962 |        |       |       |    c2 = xp.float32(0.5) * dt * (xp.float32(2.0) + cnm1)                                                   |
|   963 |        |       |       |    c3 = -xp.float32(0.5) * dt * cnm1                                                                      |
|   964 |        |       |       |                                                                                                           |
|   965 |        |       |       |    NUM = S.step3_NUM                                                                                      |
|   966 |        |       |       |    NUM[...] = om2                                                                                         |
|   967 |        |       |   2%  |    xp.multiply(om2, ARG, out=tmp_c)                                                                       |
|   968 |        |       |       |    NUM -= tmp_c                                                                                           |
